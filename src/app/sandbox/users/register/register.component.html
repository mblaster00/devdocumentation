Users


This operation is used to register as a client of Logidoo Ecommerce API.

/register - POST


Try it

Request URL

Request headers

  • string

    Basic authentication header containing API user ID and API key. Should be sent in as HS245 encoded.

  • (optional)
    string
    Media type of the body sent to the API.

Request body

  • application/x-www-form-urlencoded
access_token

curl --location --request POST 'http://localhost:9000/api/users/register' \
--header 'Content-Type: application/x-www-form-urlencoded' \
--data-urlencode 'username=user' \
--data-urlencode 'email=user@gmail.com' \
--data-urlencode 'phone=+221781234568' \
--data-urlencode 'companyName=Logidoo' \
--data-urlencode 'country=Senegal' \
--data-urlencode 'password=pass123'
                

var client = new RestClient("http://localhost:9000/api/users/register");
client.Timeout = -1;
var request = new RestRequest(Method.POST);
request.AddHeader("Content-Type", "application/x-www-form-urlencoded");
request.AddParameter("username", "user");
request.AddParameter("email", "user@gmail.com");
request.AddParameter("phone", "+221781234568");
request.AddParameter("companyName", "Logidoo");
request.AddParameter("country", "Senegal");
request.AddParameter("password", "pass123");
IRestResponse response = client.Execute(request);
Console.WriteLine(response.Content);
                

OkHttpClient client = new OkHttpClient().newBuilder()
  .build();
MediaType mediaType = MediaType.parse("application/x-www-form-urlencoded");
RequestBody body = RequestBody.create(mediaType, "username=user&email=user@gmail.com&phone=+221781234568&companyName=Logidoo&country=Senegal&password=pass123");
Request request = new Request.Builder()
  .url("http://localhost:9000/api/users/register")
  .method("POST", body)
  .addHeader("Content-Type", "application/x-www-form-urlencoded")
  .build();
Response response = client.newCall(request).execute();
